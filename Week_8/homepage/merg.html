                void MergeSort(vector<int>& X, int l, int r, int n) {
                    if (n > 2) {
                        MergeSort(X, l, (r + l) / 2, n / 2 + n % 2);
                        MergeSort(X, (r + l) / 2 + 1, r, n / 2);
                    }

                    if (n == 1)
                        return;

                    vector<int> tmp(n);
                    int k = 0, l1 = l, l2 = (r + l) / 2 + 1;
                    int n1 = l2, n2 = n + l;

                    while (l1 < n1 && l2 < n2) {
                        if (X[l1] < X[l2]) {
                            tmp[k] = X[l1];
                            l1++, k++;
                        } else {
                            tmp[k] = X[l2];
                            l2++, k++;
                        }
                    }

                    while (l1 < n1) {
                        tmp[k] = X[l1];
                        l1++, k++;
                    }

                    while (l2 < n2) {
                        tmp[k] = X[l2];
                        l2++, k++;
                    }

                    for (int i = 0; i < n; i++, l++)
                        X[l] = tmp[i];
                }

                bool Binary_search(vector<int> &X, int n) {
                    int l = 0, r = X.size() - 1 , at;
                    while (l < r) {
                        at = (l + r) / 2;
                        if (X[at] < n) {
                            l = at + 1;
                        } else {
                            r = at;
                        }
                    }
                    return(X[l] == n);
                }
